            rows.forEach(row => {
                const appName = row.querySelector('td:first-child').textContent.toLowerCase();

                    row.style.display = '';
                } else {
                    row.style.display = 'none';
                }
            });
        });
    }

            else if (pageId === 'statistic') loadStatisticsPage();

        generateRandomCode: () => {
            const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789';
            let code = '';
            for (let i = 0; i < 5; i++) {
                code += chars.charAt(Math.floor(Math.random() * chars.length));
            }
            return code;
        },

        formatDate: (dateString) => {
            const date = new Date(dateString);
            return date.toLocaleString('ru-RU');
        },

        openModal: (modalId) => {
            document.getElementById(modalId).style.display = 'flex';
        },

        closeModal: (modalId) => {
            document.getElementById(modalId).style.display = 'none';
        },

        getInitials: (name) => {
            if (!name) return 'U';
            const names = name.split(' ');
            let initials = names[0][0];
            if (names.length > 1) initials += names[1][0];
            return initials.toUpperCase();
        },

        formatTime: (minutes) => {
            const hours = Math.floor(minutes / 60);
            const mins = minutes % 60;
            return `${hours > 0 ? hours + 'Ñ‡ ' : ''}${mins}Ð¼Ð¸Ð½`;
        }
    };

    // Data loading for pages
    function loadDashboard() {
        const user = userService.getCurrentUser();
        if (!user) return;

        const devices = deviceService.getUserDevices(user.id);
        DOM.devicesGrid.innerHTML = '';

        // update the statistic
        DOM.devicesCount.textContent = devices.length;

        let blockedCount = 0;
        let totalScreenTime = 0;

        devices.forEach(device => {
            if (device.isBlocked) {
                blockedCount++;
            }
            totalScreenTime += device.screenTime || 0;
        });

        DOM.blockedCount.textContent = blockedCount;

        const avgScreenTime = devices.length > 0 ? Math.round(totalScreenTime / devices.length) : 0;
        DOM.screenTimeAvg.textContent = `${avgScreenTime} min`;

        if (devices.length === 0) {
            DOM.devicesGrid.innerHTML = `
                      <div class="empty-state">
                          <i class="fas fa-mobile-alt"></i>
                          <h3>Devices Were Not Found</h3>
                          <p>You Have Not Added Any Tracking Devices Yet</p>
                          <button id="addFirstDeviceBtn" class="btn btn-primary" style="margin-top: 20px;">
                              <i class="fas fa-plus"></i> Add First Device
                          </button>
                      </div>
                  `;

            document.getElementById('addFirstDeviceBtn').addEventListener('click', () => {
                DOM.addDeviceBtn.click();
            });
        } else {
            devices.forEach(device => {
                const lastActive = new Date(device.lastActive);
                const now = new Date();
                const minutesAgo = Math.floor((now - lastActive) / 60000);

                // Counting of percent for screen time usage
                const screenTimePercent = device.screenTimeLimit > 0 ?
                    Math.min(100, Math.round((device.screenTime || 0) / device.screenTimeLimit * 100)) :
                    0;

                // Get device icon based on type
                let deviceIcon = 'fa-mobile-alt';
                if (device.deviceType === 'tablet') deviceIcon = 'fa-tablet-alt';
                else if (device.deviceType === 'laptop') deviceIcon = 'fa-laptop';
                else if (device.devæžType === 'desktop') deviceIcon = 'fa-desktop';

                const deviceCard = document.createElement('div');
                deviceCard.className = 'device-card';
                deviceCard.innerHTML = `
                          <div class="device-status ${minutesAgo > 5 ? 'offline' : ''} ${device.isBlocked ? 'offline' : ''}"></div>
                          <div class="device-name">
                              <i class="fas ${deviceIcon}"></i> ${device.name}
                              ${device.isBlocked ? '<span style="color: var(--danger); margin-left: 10px;"><i class="fas fa-lock"></i> Is Blocked</span>' : ''}
                          </div>
                          <div class="device-info">
                              <span>${device.model}</span>
                              <span>${device.osVersion}</span>
                              <span>Status: ${minutesAgo > 5 ? 'Offline' : 'Online'}</span>
                              <span>Last Activity: ${minutesAgo} minutes ago</span>
                          </div>

                          <div style="margin: 15px 0;">
                              <div style="display: flex; justify-content: space-between;">
                                  <span>Screen Time:</span>
                                  <span><strong>${utils.formatTime(device.screenTime || 0)}</strong> / ${utils.formatTime(device.screenTimeLimit)}</span>
                              </div>
                              <div class="screen-time-progress">
                                  <æž class="screen-time-bar" style="width: ${screenTimePercent}%"></div>
                              </div>
                          </div>

                          <div class="device-actions">
                              <button class="btn btn-sm ${device.isBlocked ? 'btn-success' : 'btn-danger'}"
                                      data-device="${device.id}" data-action="${device.isBlocked ? 'unblock' : 'block'}">
                                  <i class="fas ${device.isBlocked ? 'fa-lock-open' : 'fa-lock'}"></i>
                                  ${device.isBlocked ? 'Unblock' : 'Block'}
                              </button>
                              <button class="btn btn-sm btn-outline" data-device="${device.id}" data-action="screenshot">
                                  <i class="fas fa-camera"></i> Screenshot
                              </button>
                          </div>
                      `;

                DOM.devicesGrid.appendChild(deviceCard);
            });

            // Handlers for control buttons
            document.querySelectorAll('[data-action="block"]').forEach(btn => {
                btn.addEventListener('click', () => {
                    const deviceId = btn.getAttribute('data-device');
                    deviceService.blockDevice(deviceId);
                    utils.showNotification('The Device Is Blocked');
                    loadDashboard();
                });
            });

            document.querySelectorAll('[data-action="unblock"]').forEach(btn => {
                btn.addEventListener('click', () => {
                    const deviceId = btn.getAttribute('data-device');
                    deviceService.unblockDevice(deviceId);
                    utils.showNotification('The Device Was Unblocked');
                    loadDashboard();
                });
            });

            document.querySelectorAll('[data-action="screenshot"]').forEach(btn => {
                btn.addEventListener('click', () => {
                    const deviceId = btn.getAttribute('data-device');
                    screenshotService.addScreenshot(deviceId);
                    utils.showNotification('Screenshot Request Was Sent');
                    utils.navigateToPage('monitoring');

                            <i class="fas fa-plus"></i> Add First Device
                        </button>
                    </div>
                `;

            document.getElementById('addFirstDeviceBtnManage').addEventListener('click', () => {
                DOM.addDeviceBtn.click();
            });
        } else {
            devices.forEach(device => {
                // Get device icon based on type
                let deviceIcon = 'fa-mobile-alt';
                if (device.deviceType === 'tablet') deviceIcon = 'fa-tablet-alt';
                else if (device.deviceType === 'laptop') deviceIcon = 'fa-laptop';
                else if (device.deviceType === 'desktop') deviceIcon = 'fa-desktop';

                const deviceCard = document.createElement('div');
                deviceCard.className = 'device-card';
                deviceCard.innerHTML = `
                        <div class="device-name">
                            <i class="fas ${deviceIcon}"></i> ${device.name}
                            ${device.isBlocked ? '<span style="color: var(--danger); margin-left: 10px;"><i class="fas fa-lock"></i> Is Blocked</span>' : ''}
                        </div>
                        <div class="device-info">
                            ${device.model} | ${device.osVersion}
                        </div>
                        <div class="device-actions">
                            <button class="btn btn-sm ${device.isBlocked ? 'btn-success' : 'btn-danger'}"
                                    data-device="${device.id}" data-action="${device.isBlocked ? 'unblock' : 'block'}">
                                <i class="fas ${device.isBlocked ? 'fa-lock-open' : 'fa-lock'}"></i>
                                ${device.isBlocked ? 'Unblock' : 'Block'}
                            </button>
                            <button class="btn btn-sm btn-outline" data-device="${device.id}" data-action="categories">
                                <i class="fas fa-layer-group"></i> Categories
                            </button>
                        </div>
                    `;

                DOM.manageDevicesGrid.appendChild(deviceCard);
            });

            // Handlers for managing buttons
            document.querySelectorAll('[data-action="block"]').forEach(btn => {
                btn.addEventListener('click', () => {
                    const deviceId = btn.getAttribute('data-device');
                    deviceService.blockDevice(deviceId);
                    utils.showNotification('Device Is Blocked');
                    loadDevicesPage();
                });
            });

            document.querySelectorAll('[data-action="unblock"]').forEach(btn => {
                btn.addEventListener('click', () => {
                    const deviceId = btn.getAttribute('data-device');
                    deviceService.unblockDevice(deviceId);
                    utils.showNotification('Device Is Unblocked');
                    loadDevicesPage();
                });
            });

            document.querySelectorAll('[data-action="categories"]').forEach(btn => {
                btn.addEventListener('click', () => {
                    utils.navigateToPage('monitoring');
                });
            });
        }

        // Tethered code generation
        DOM.bindingCode.textContent = utils.generateRandomCode();
    }

    function loadMonitoringPage() {
        const user = userService.getCurrentUser();
        if (!user) return;

        const devices = deviceService.getUserDevices(user.id);
        const selectedDeviceId = DOM.monitoringDeviceSelect.value;

        // update the list of devices
        DOM.monitoringDeviceSelect.innerHTML = '<option value="all">All Devices</option>';
        devices.forEach(device => {
            const option = document.createElement('option');
            option.value = device.id;
            option.textContent = device.name;
            if (device.id === selectedDeviceId) option.selected = true;
            DOM.monitoringDeviceSelect.appendChild(option);
        });

        // Render categories
        renderCategoriesBar();

        // screenshots loading
        let screenshots = [];

        if (selectedDeviceId === 'all') {
            screenshots = screenshotService.getAllScreenshots()
                .filter(s => devices.some(d => d.id === s.deviceId));
        } else {
            screenshots = screenshotService.getDeviceScreenshots(selectedDeviceId);
        }

        DOM.monitoringScreenshotsContainer.innerHTML = '';

        if (screenshots.length === 0) {
            DOM.monitoringScreenshotsContainer.innerHTML = `
                    <div class="empty-state" style="grid-column: 1/-1;">
                        <i class="fas fa-camera"></i>
                        <h3>No Screenshots</h3>
                        <p>You Have Not Taken Screenshots or Tethered Your Device Yet</p>
                        <button id="requestFirstScreenshotBtn" class="btn btn-primary" style="margin-top: 20px;">
                            <i class="fas fa-camera"></i> Request First Screenshot
                        </button>
                    </div>
                `;

            document.getElementById('requestFirstScreenshotBtn').addEventListener('click', () => {
                DOM.newScreenshotBtn.click();
            });
        } else {
            screenshots.forEach(screenshot => {
                const device = deviceService.getDevice(screenshot.deviceId);

                const screenshotEl = document.createElement('div');
                screenshotEl.className = 'screenshot-item';
                screenshotEl.dataset.screenshotId = screenshot.id;
                screenshotEl.innerHTML = `
                        <img src="${screenshot.url}" alt="Screenshot">
                        <div class="screenshot-info">
                            <div class="device-name">${device.name}</div>
                            <div class="screenshot-date">${utils.formatDate(screenshot.createdAt)}</div>
                            <div>Category: Games</div>
                            <div class="screenshot-actions">
                                <button class="btn btn-sm btn-outline" data-action="view">
                                    <i class="fas fa-eye"></i> Viewing
                                </button>
                                <button class="btn btn-sm btn-outline" data-action="delete">
                                    <i class="fas fa-trash"></i> Delete
                                </button>
                            </div>
                        </div>
                    `;

                DOM.monitoringScreenshotsContainer.appendChild(screenshotEl);

                // Handler for screenshot viewing
                screenshotEl.querySelector('[data-action="view"]').addEventListener('click', (e) => {
                    e.stopPropagation();
                    document.getElementById('screenshotImage').src = screenshot.url;
                    document.getElementById('screenshotDeviceName').textContent = device.name;
                    document.getElementById('screenshotDate').textContent = utils.formatDate(screenshot.createdAt);
                    document.getElementById('deleteScreenshotBtn').dataset.screenshotId = screenshot.id;
                    utils.openModal('screenshotModal');
                });

                // Handler for screenshot deleting
                screenshotEl.querySelector('[data-action="delete"]').addEventListener('click', (e) => {
                    e.stopPropagation();
                    if (confirm('Are You Sure that You Want to Delete this Screenshot?')) {
                        screenshotService.deleteScreenshot(screenshot.id);
                        utils.showNotification('Screenshot Was Deleted Successfully.');
                        loadMonitoringPage();
                    }
                });
            });
        }
    }

    function loadSettingsPage() {
        const settings = JSON.parse(localStorage.getItem(DB.SETTINGS));

        // Fill in settings fields
        document.getElementById('screenTimeLimit').value = settings.screenTimeLimit;
        document.getElementById('timeStart').value = settings.timeStart;
        document.getElementById('timeEnd').value = settings.timeEnd;
        document.getElementById('screenshotFrequency').value = settings.screenshotFrequency;
        document.getElementById('screenshotQuality').value = settings.screenshotQuality;
        document.getElementById('autoLock').value = settings.autoLock;
    }

    function loadUserInfo() {
        const user = userService.getCurrentUser();
        if (!user) return;

        DOM.userInfo.innerHTML = `
                <div class="user-avatar">${utils.getInitials(user.name)}</div>
                <div>${user.name}</div>
            `;
    }

    // Handler for deleting a screenshot in a modal window
    document.getElementById('deleteScreenshotBtn').addEventListener('click', function() {
        const screenshotId = this.dataset.screenshotId;
        if (screenshotId && confirm('Are you sure you want to delete this screenshot?')) {
            screenshotService.deleteScreenshot(screenshotId);
            utils.showNotification('Screenshot Was Deleted');
            utils.closeModal('screenshotModal');
            loadMonitoringPage();
        }
    });
